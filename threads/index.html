> curl 'https://api.m3o.com/threads/Threads/CreateMessage' \
  -H 'micro-namespace: $yourNamespace' \
  -H 'authorization: Bearer $yourToken' \
  -d {
  "author_id": "string",
  "id": "string",
  "text": "string",
  "thread_id": "string"
};
# Response
{
  "message": {
    "author_id": "string",
    "id": "string",
    "sent_at": "string",
    "text": "string",
    "thread_id": "string"
  }
}
