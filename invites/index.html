> curl 'https://api.m3o.com/protobuf/Invites/Create' \
  -H 'micro-namespace: $yourNamespace' \
  -H 'authorization: Bearer $yourToken' \
  -d {
  "email": "string",
  "group_id": "string"
};
# Response
{
  "invite": {
    "code": "string",
    "email": "string",
    "group_id": "string",
    "id": "string"
  }
}
